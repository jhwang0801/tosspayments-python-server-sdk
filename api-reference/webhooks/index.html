    WebhookResource

    The WebhookResource class handles webhook event verification and parsing for real-time payment notifications from TossPayments.

    Class Overview

    class WebhookResource:
    """Webhook verification and parsing for TossPayments events"""

    Access the WebhookResource through the client:

    from tosspayments_server_sdk import Client

client = Client(secret_key="test_sk_...")
webhooks = client.webhooks  # WebhookResource instance

    Methods

    verify_and_parse()

    Verify and parse webhook payload from TossPayments.

    def verify_and_parse(self, payload: Union[str, bytes]) -> WebhookEvent

    Parameters:

    Parameter Type Required Description
    payload Union[str, bytes] Yes Raw webhook payload from TossPayments

    Returns: WebhookEvent - Parsed webhook event object

    Raises: - WebhookVerificationError - If webhook verification fails - ValueError - If payload format is invalid

    Example: 

    from tosspayments_server_sdk import WebhookVerificationError

@app.route('/webhook', methods=['POST'])
def handle_webhook():
    try:
        # Parse webhook data
        webhook_event = client.webhooks.verify_and_parse(request.data)

        # Handle different event types
        if webhook_event.is_payment_event:
            handle_payment_event(webhook_event)
        elif webhook_event.is_cancel_event:
            handle_cancel_event(webhook_event)
        elif webhook_event.is_virtual_account_event:
            handle_virtual_account_event(webhook_event)

        return "OK", 200

    except WebhookVerificationError as e:
        print(f"Webhook verification failed: {e}")
        return "Bad Request", 400

    Webhook Events

    WebhookEvent Type Union

    WebhookEvent = Union[
    PaymentStatusChangedEvent,
    DepositCallbackEvent,

    All webhook events inherit from BaseWebhookEvent and provide common properties.

    BaseWebhookEvent

    Base class for all webhook events.

    @dataclass
class BaseWebhookEvent(BaseModel, ABC):
    created_at: datetime
    event_type: str

    Properties

    @property
def is_payment_event(self) -> bool
    Returns True if this is a payment status change event.

    @property
def is_cancel_event(self) -> bool
    Returns True if this is a cancellation event.

    @property
def is_virtual_account_event(self) -> bool
    Returns True if this is a virtual account deposit event.

    PaymentStatusChangedEvent

    Triggered when a payment status changes.

    @dataclass
class PaymentStatusChangedEvent(BaseWebhookEvent):
    payment: Payment

    Additional Properties

    @property
def payment_key(self) -> str
    Returns the payment key from the embedded payment object.

    @property
def order_id(self) -> str
    Returns the order ID from the embedded payment object.

    @property
def status(self) -> PaymentStatus
    Returns the current payment status.

    Methods

    def is_payment_completed(self) -> bool
    Returns True if the payment status is DONE.

    def is_payment_canceled(self) -> bool
    Returns True if the payment status is CANCELED or PARTIAL_CANCELED.

    Complete Webhook Handler Example

    Flask Integration

    from flask import Flask, request
from tosspayments_server_sdk import Client, WebhookVerificationError

app = Flask(__name__)
client = Client(secret_key=os.getenv("TOSS_SECRET_KEY"))

@app.route('/webhook', methods=['POST'])
def toss_webhook():
    try:
        # Parse webhook event
        event = client.webhooks.verify_and_parse(request.data)

        # Route to appropriate handler
        if event.is_payment_event:
            result = handle_payment_event(event)
        elif event.is_cancel_event:
            result = handle_cancel_event(event)
        elif event.is_virtual_account_event:
            result = handle_virtual_account_event(event)
        else:
            print(f"Unknown event type: {event.event_type}")
            return "Unknown Event", 400

        # Log successful processing
        print(f"Successfully processed {event.event_type} event")
        return "OK", 200

    except WebhookVerificationError as e:
        print(f"Webhook verification failed: {e}")
        return "Verification Failed", 400

    except Exception as e:
        print(f"Webhook processing error: {e}")
        return "Internal Error", 500

def handle_payment_event(event):
    """Handle payment status change events"""
    payment = event.payment

    if event.is_payment_completed():
        # Payment completed successfully
        update_order_status(payment.order_id, "completed")
        send_confirmation_email(payment.order_id)
        update_inventory(payment.order_id)

    elif event.is_payment_canceled():
        # Payment was canceled
        update_order_status(payment.order_id, "canceled")
        process_refund(payment.order_id)

    return {"status": "processed", "payment_key": payment.payment_key}

def handle_cancel_event(event):
    """Handle cancellation events"""
    # Process the cancellation
    process_cancellation(
        transaction_key=event.transaction_key,
        amount=event.cancel_amount,
        reason=event.cancel_reason
    )

    return {"status": "processed", "transaction_key": event.transaction_key}

def handle_virtual_account_event(event):
    """Handle virtual account deposit events"""
    if event.is_deposit_completed():
        # Virtual account deposit completed
        update_order_status(event.order_id, "paid")
        process_order_fulfillment(event.order_id)

    return {"status": "processed", "order_id": event.order_id}

    Webhook Security

    Verification Process

    The SDK automatically handles webhook verification to ensure the request comes from TossPayments:

    1. Signature Verification: Validates the webhook signature
    2. Payload Integrity: Ensures payload hasn't been tampered with
    3. Origin Validation: Confirms the request originates from TossPayments

    Best Practices

    def secure_webhook_handler(request):
    try:
        # Always verify webhook first
        event = client.webhooks.verify_and_parse(request.body)

        # Implement idempotency
        if is_duplicate_event(event):
            return "Already Processed", 200

        # Process event atomically
        with database.transaction():
            process_webhook_event(event)
            mark_event_processed(event)

        return "OK", 200

    except WebhookVerificationError:
        # Log security incident
        log_security_incident(request)
        return "Unauthorized", 401

    Idempotency Handling

    def is_duplicate_event(event):
    """Check if we've already processed this event"""
    if hasattr(event, 'payment_key'):
        return EventLog.objects.filter(
            payment_key=event.payment_key,
            event_type=event.event_type,
            created_at=event.created_at
        ).exists()
    return False

def mark_event_processed(event):
    """Mark event as processed to prevent duplicates"""
    EventLog.objects.create(
        payment_key=getattr(event, 'payment_key', None),
        order_id=getattr(event, 'order_id', None),
        event_type=event.event_type,
        created_at=event.created_at,
        processed_at=timezone.now()
    )

    Error Handling

    Common Webhook Errors

    from tosspayments_server_sdk import WebhookVerificationError

def robust_webhook_handler(request):
    try:
        event = client.webhooks.verify_and_parse(request.body)
        return process_webhook_event(event)

    except WebhookVerificationError as e:
        # Invalid signature or tampered payload
        logger.warning(f"Webhook verification failed: {e}")
        return HttpResponse("Verification Failed", status=400)

    except ValueError as e:
        # Invalid JSON or payload format
        logger.error(f"Invalid webhook payload: {e}")
        return HttpResponse("Invalid Payload", status=400)

    except Exception as e:
        # Unexpected error
        logger.error(f"Webhook processing error: {e}")
        return HttpResponse("Internal Error", status=500)

    Retry Logic

    TossPayments will retry failed webhooks, so ensure your handler is idempotent:

    def idempotent_webhook_handler(event):
    """Process webhook in an idempotent manner"""

    # Check if already processed
    if is_event_processed(event):
        return "Already Processed"

    try:
        # Process the event
        result = process_event(event)

        # Mark as processed
        mark_event_processed(event)

        return result

    except Exception as e:
        # Don't mark as processed if it failed
        logger.error(f"Event processing failed: {e}")
        raise
