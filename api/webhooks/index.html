    웹훅 (Webhook)

    토스페이먼츠에서 결제 상태 변경, 가상계좌 입금 등의 이벤트를 실시간으로 알려주는 웹훅을 처리하는 방법을 안내합니다.

    웹훅이란?

    웹훅은 토스페이먼츠에서 결제 상태가 변경될 때 여러분의 서버로 실시간 알림을 보내는 기능입니다. 이를 통해:

    • 결제 완료/실패를 즉시 감지
    • 가상계좌 입금 확인
    • 결제 취소 처리
    • 자동화된 주문 처리

    지원하는 웹훅 이벤트

    1. 결제 상태 변경 (PAYMENT_STATUS_CHANGED)

    결제 상태가 변경될 때 발생:

    {
  "eventType": "PAYMENT_STATUS_CHANGED",
  "createdAt": "2024-01-01T12:00:00.000Z",
  "data": {
    "paymentKey": "mc_1234567890123456789012345678901234567890",
    "orderId": "order-20240101-001",
    "status": "DONE",
    ...
  }
}

    2. 취소 상태 변경 (CANCEL_STATUS_CHANGED)

    결제 취소가 완료될 때 발생:

    {
  "eventType": "CANCEL_STATUS_CHANGED", 
  "createdAt": "2024-01-01T12:00:00.000Z",
  "data": {
    "cancelAmount": 5000,
    "cancelReason": "단순 변심",
    "canceledAt": "2024-01-01T12:00:00.000Z",
    ...
  }
}

    3. 가상계좌 입금 콜백 (DEPOSIT_CALLBACK)

    가상계좌에 입금이 완료될 때 발생:

    {
  "createdAt": "2024-01-01T12:00:00.000Z",
  "secret": "webhook_secret_value",
  "status": "DONE",
  "transactionKey": "9FF15E1A29D0E77C218F57262BFA4986",
  "orderId": "order-20240101-001"
}

client = Client(secret_key="your_secret_key")

webhook_event = client.webhooks.verify_and_parse(request_body)

# 이벤트 타입 확인
if webhook_event.is_payment_event:
    print("결제 상태 변경 이벤트")
elif webhook_event.is_cancel_event:
    print("취소 상태 변경 이벤트")
elif webhook_event.is_virtual_account_event:
    print("가상계좌 입금 이벤트")

    프레임워크별 구현

    Flask 예시

    from flask import Flask, request, jsonify
from tosspayments_server_sdk import Client
from tosspayments_server_sdk.exceptions import WebhookVerificationError

app = Flask(__name__)
client = Client(secret_key="your_secret_key")

@app.route('/webhook/toss', methods=['POST'])
def handle_toss_webhook():
    try:
        # 웹훅 파싱
        webhook_event = client.webhooks.verify_and_parse(request.data)

        # 이벤트 타입별 처리
        if webhook_event.is_payment_event:
            return handle_payment_event(webhook_event)
        elif webhook_event.is_cancel_event:
            return handle_cancel_event(webhook_event)
        elif webhook_event.is_virtual_account_event:
            return handle_deposit_event(webhook_event)

        return jsonify({"status": "unknown_event"}), 400

    except WebhookVerificationError as e:
        print(f"웹훅 검증 실패: {e}")
        return jsonify({"error": "invalid_webhook"}), 400
    except Exception as e:
        print(f"웹훅 처리 오류: {e}")
        return jsonify({"error": "internal_error"}), 500

def handle_payment_event(event):
    payment = event.payment

    if payment.is_paid():
        # 결제 완료 처리
        complete_order(payment.order_id, payment)

    elif payment.is_canceled():
        # 결제 취소 처리
        cancel_order(payment.order_id, payment)

    return jsonify({"status": "success"})

def handle_cancel_event(event):
    cancellation = event.cancellation

    # 취소 완료 처리
    process_cancellation(cancellation)

    return jsonify({"status": "success"})

def handle_deposit_event(event):
    if event.is_deposit_completed():
        # 가상계좌 입금 완료 처리
        complete_virtual_account_payment(event.order_id)

    return jsonify({"status": "success"})

    # views.py
from django.http import JsonResponse
from django.views.decorators.csrf import csrf_exempt
from django.views.decorators.http import require_http_methods

@csrf_exempt
@require_http_methods(["POST"])
def toss_webhook(request):

            payment = webhook_event.payment

            # 주문 상태 업데이트
            from myapp.models import Order
            order = Order.objects.get(order_id=payment.order_id)

            if payment.is_paid():
                order.status = 'paid'
                order.payment_key = payment.payment_key
                order.save()

                # 결제 완료 후 처리 (이메일 발송, 재고 차감 등)
                order.process_payment_completion()

        return JsonResponse({"status": "success"})

    except Exception as e:
        return JsonResponse({"error": str(e)}, status=400)

    가상계좌 웹훅 검증

    가상계좌 입금 웹훅의 경우 secret 값으로 검증:

    from tosspayments_server_sdk.utils import verify_payment_webhook_secret

def handle_deposit_event(event):
    # secret 검증 (결제 승인 시 받은 secret과 비교)
    expected_secret = get_payment_secret(event.order_id)  # DB에서 조회

    webhook_data = {
        "secret": event.secret,
        "status": event.status.value,
        "transactionKey": event.transaction_key,
        "orderId": event.order_id
    }

    if verify_payment_webhook_secret(webhook_data, expected_secret):
        # 검증 성공
        process_deposit(event)
    else:
        # 검증 실패
        print("웹훅 검증 실패")
        return False

    멱등성 처리

    웹훅은 재시도될 수 있으므로 멱등성을 보장해야 합니다:

    def handle_payment_event(event):
    payment = event.payment

    # 이미 처리된 웹훅인지 확인
    if is_webhook_already_processed(payment.payment_key, event.created_at):
        return {"status": "already_processed"}

    # 웹훅 처리
    if payment.is_paid():
        complete_payment(payment)

    # 처리 완료 기록
    mark_webhook_as_processed(payment.payment_key, event.created_at)

    return {"status": "success"}

def is_webhook_already_processed(payment_key, created_at):
    # DB에서 처리 여부 확인
    from myapp.models import WebhookLog
    return WebhookLog.objects.filter(
        payment_key=payment_key,
        created_at=created_at
    ).exists()

    모범 사례

    1. 응답 시간 최적화

    웹훅 처리는 5초 이내에 완료해야 합니다:

    from concurrent.futures import ThreadPoolExecutor

executor = ThreadPoolExecutor(max_workers=10)

@app.route('/webhook/toss', methods=['POST'])
def handle_webhook():
    try:
        webhook_event = client.webhooks.verify_and_parse(request.data)

        # 즉시 응답 후 백그라운드에서 처리
        executor.submit(process_webhook_async, webhook_event)

        return {"status": "received"}, 200

    except Exception as e:
        return {"error": str(e)}, 400

def process_webhook_async(webhook_event):
    """백그라운드에서 실제 웹훅 처리"""
    try:
        if webhook_event.is_payment_event:
            handle_payment_event(webhook_event)
    except Exception as e:
        logger.error(f"비동기 웹훅 처리 실패: {e}")

    2. 데이터베이스 연결 관리

    from contextlib import contextmanager

def retry_on_failure(max_retries=3, delay=1):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None

            for attempt in range(max_retries):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    if attempt < max_retries - 1:
                        time.sleep(delay * (2 ** attempt))  # 지수 백오프

webhook_handler = WebhookHandler()

@app.route('/webhook/toss', methods=['POST'])
def toss_webhook():
    start_time = datetime.now()

    try:
        # 웹훅 파싱
        webhook_event = client.webhooks.verify_and_parse(request.data)

        # 중복 확인
        if webhook_handler.is_duplicate(webhook_event):
            logger.info("중복 웹훅 무시")
            return jsonify({"status": "duplicate"}), 200

        # 이벤트 타입별 처리
        if webhook_event.is_payment_event:
            webhook_handler.handle_payment_status_changed(webhook_event)
        elif webhook_event.is_cancel_event:
            webhook_handler.handle_cancel_status_changed(webhook_event)
        elif webhook_event.is_virtual_account_event:
            webhook_handler.handle_deposit_callback(webhook_event)
        else:
            logger.warning(f"알 수 없는 웹훅 타입: {webhook_event.event_type}")
            return jsonify({"error": "unknown_event_type"}), 400

        # 처리 시간 로깅
        process_time = (datetime.now() - start_time).total_seconds()
        logger.info(f"웹훅 처리 완료 ({process_time:.2f}초)")

        return jsonify({"status": "success"}), 200

    except WebhookVerificationError as e:
        logger.error(f"웹훅 검증 실패: {e}")
        return jsonify({"error": "verification_failed"}), 400
    except Exception as e:
        logger.error(f"웹훅 처리 오류: {e}", exc_info=True)

        # 에러 알림
        send_error_notification(str(e), request.data)

        return jsonify({"error": "internal_error"}), 500
