    에러 처리

    TossPayments SDK에서 발생할 수 있는 다양한 에러를 처리하는 방법을 안내합니다.

    에러 종류

    SDK는 다음과 같은 예외 클래스들을 제공합니다:

    기본 예외 계층

    TossPaymentsError (기본 예외)
├── APIError (API 호출 관련)
│   ├── AuthenticationError (인증 오류)
│   ├── RateLimitError (요청 한도 초과)
│   └── PaymentNotFoundError (결제 정보 없음)
├── ValidationError (입력값 검증 오류)
├── WebhookVerificationError (웹훅 검증 오류)
└── NetworkError (네트워크 오류)

    예외 클래스별 상세

    1. APIError - API 호출 오류

    토스페이먼츠 API 호출 시 발생하는 모든 오류의 기본 클래스입니다.

    from tosspayments_server_sdk.exceptions import APIError

try:
    payment = client.payments.confirm(
        payment_key="invalid_key",
        order_id="order-123", 
        amount=15000
    )
except APIError as e:
    print(f"API 에러: {e.message}")
    print(f"에러 코드: {e.error_code}")
    print(f"HTTP 상태코드: {e.status_code}")
    print(f"응답 데이터: {e.response_data}")

    주요 속성: - message: 에러 메시지 - error_code: 토스페이먼츠 에러 코드 - status_code: HTTP 상태 코드 - response_data: API 응답 전체 데이터

    2. AuthenticationError - 인증 오류

    API 키가 잘못되었거나 인증에 실패한 경우입니다.

    from tosspayments_server_sdk.exceptions import AuthenticationError

try:
    client = Client(secret_key="wrong_key")
    payment = client.payments.retrieve("payment_key")
except AuthenticationError as e:
    print(f"인증 실패: {e}")
    # API 키 확인 필요

    일반적인 원인: - 잘못된 API 키 형식 - 만료된 API 키
    - 권한이 없는 API 키

    3. ValidationError - 입력값 검증 오류

    SDK에 전달된 매개변수가 유효하지 않은 경우입니다.

    from tosspayments_server_sdk.exceptions import ValidationError

try:
    # 잘못된 입력값
    payment = client.payments.confirm(
        payment_key="",  # 빈 문자열
        order_id="order-123",
        amount=-1000  # 음수 금액
    )
except ValidationError as e:
    print(f"입력값 오류: {e}")

    일반적인 원인: - 필수 매개변수 누락 - 잘못된 데이터 타입 - 범위를 벗어난 값

    4. PaymentNotFoundError - 결제 정보 없음

    존재하지 않는 결제를 조회하려 할 때 발생합니다.

    from tosspayments_server_sdk.exceptions import PaymentNotFoundError

try:
    payment = client.payments.retrieve("non_existent_payment_key")
except PaymentNotFoundError as e:
    print(f"결제 정보 없음: {e}")
    # 주문 ID나 결제 키 확인 필요

    5. RateLimitError - 요청 한도 초과

    API 호출 횟수가 제한을 초과한 경우입니다.

    from tosspayments_server_sdk.exceptions import RateLimitError
import time

try:
    payment = client.payments.retrieve("payment_key")
except RateLimitError as e:
    print(f"요청 한도 초과: {e}")
    # 잠시 대기 후 재시도
    time.sleep(5)
    payment = client.payments.retrieve("payment_key")

    6. NetworkError - 네트워크 오류

    인터넷 연결 문제나 서버 응답 없음 등의 네트워크 관련 오류입니다.

    from tosspayments_server_sdk.exceptions import NetworkError

try:
    payment = client.payments.retrieve("payment_key")
except NetworkError as e:
    print(f"네트워크 오류: {e}")
    # 네트워크 상태 확인 또는 재시도

    7. WebhookVerificationError - 웹훅 검증 오류

    웹훅 데이터 파싱에 실패한 경우입니다.

    from tosspayments_server_sdk.exceptions import WebhookVerificationError

try:
    webhook_event = client.webhooks.verify_and_parse(invalid_data)
except WebhookVerificationError as e:
    print(f"웹훅 검증 실패: {e}")
    # 웹훅 데이터 형식 확인

    except Exception as e:
        # 예상치 못한 오류
        logger.error(f"예상치 못한 오류: {e}", exc_info=True)
        return {"success": False, "error": "시스템 오류가 발생했습니다"}

    2. 구체적인 에러별 처리

    def process_payment(payment_key, order_id, amount):
    try:
        payment = client.payments.confirm(
            payment_key=payment_key,
            order_id=order_id, 
            amount=amount
        )
        return handle_successful_payment(payment)

    except ValidationError as e:
        # 입력값 오류 - 클라이언트에 상세 정보 제공
        return {
            "success": False,
            "error_type": "validation_error",
            "message": f"입력값 오류: {e}",
            "retry_possible": False
        }

    except AuthenticationError as e:
        # 인증 오류 - 시스템 관리자에게 알림
        logger.critical(f"API 인증 실패: {e}")
        send_admin_alert("API 키 확인 필요")
        return {
            "success": False,
            "error_type": "auth_error", 
            "message": "시스템 오류가 발생했습니다",
            "retry_possible": False
        }

    except PaymentNotFoundError as e:
        # 결제 정보 없음 - 주문 상태 확인
        return {
            "success": False,
            "error_type": "payment_not_found",
            "message": "결제 정보를 찾을 수 없습니다",
            "retry_possible": False
        }

    except RateLimitError as e:
        # 요청 한도 초과 - 재시도 안내
        return {
            "success": False,
            "error_type": "rate_limit",
            "message": "잠시 후 다시 시도해주세요",
            "retry_possible": True,
            "retry_after": 60
        }

    except NetworkError as e:
        # 네트워크 오류 - 재시도 가능
        logger.warning(f"네트워크 오류: {e}")
        return {
            "success": False,
            "error_type": "network_error",
            "message": "일시적인 네트워크 오류입니다",
            "retry_possible": True
        }

    except APIError as e:
        # 기타 API 오류
        logger.error(f"API 오류 [{e.error_code}]: {e.message}")
        return {
            "success": False,
            "error_type": "api_error",
            "error_code": e.error_code,
            "message": e.message,
            "retry_possible": should_retry(e.error_code)
        }

def should_retry(error_code):
    """에러 코드에 따른 재시도 가능 여부 판단"""
    non_retryable_codes = {
        "INVALID_PAYMENT_KEY",
        "ALREADY_CANCELED_PAYMENT", 
        "INVALID_ORDER_ID"
    }
    return error_code not in non_retryable_codes

    3. 재시도 로직

    import time
from functools import wraps

def retry_on_error(max_retries=3, delay=1, backoff=2):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None

            for attempt in range(max_retries):
                try:
                    return func(*args, **kwargs)

                except (NetworkError, RateLimitError) as e:
                    last_exception = e
                    if attempt < max_retries - 1:
                        wait_time = delay * (backoff ** attempt)
                        logger.info(f"재시도 {attempt + 1}/{max_retries} ({wait_time}초 후)")
                        time.sleep(wait_time)
                    else:
                        logger.error(f"최대 재시도 횟수 초과: {e}")

                except (ValidationError, AuthenticationError, PaymentNotFoundError):
                    # 재시도해도 해결되지 않는 오류들
                    raise

            raise last_exception

        return wrapper
    return decorator

@retry_on_error(max_retries=3, delay=2)
def robust_payment_retrieve(payment_key):
    return client.payments.retrieve(payment_key)

    def handle_toss_error_code(error_code, error_message):
    """토스페이먼츠 에러 코드별 처리"""

    error_handlers = {
        # 인증 관련
        "INVALID_API_KEY": lambda: handle_auth_error("API 키가 유효하지 않습니다"),
        "UNAUTHORIZED_KEY": lambda: handle_auth_error("API 키 권한이 없습니다"),

        # 결제 관련
        "INVALID_PAYMENT_KEY": lambda: handle_payment_error("결제 키가 유효하지 않습니다"),
        "ALREADY_CANCELED_PAYMENT": lambda: handle_payment_error("이미 취소된 결제입니다"),
        "CANCEL_AMOUNT_EXCEEDED": lambda: handle_payment_error("취소 가능 금액을 초과했습니다"),

        # 주문 관련
        "INVALID_ORDER_ID": lambda: handle_order_error("주문 ID가 유효하지 않습니다"),
        "DUPLICATED_ORDER_ID": lambda: handle_order_error("중복된 주문 ID입니다"),

        # 금액 관련
        "INVALID_AMOUNT": lambda: handle_amount_error("결제 금액이 유효하지 않습니다"),
        "AMOUNT_NOT_MATCHED": lambda: handle_amount_error("결제 금액이 일치하지 않습니다"),
    }

    handler = error_handlers.get(error_code)
    if handler:
        return handler()
    else:
        return handle_unknown_error(error_code, error_message)

def handle_auth_error(message):
    logger.critical(f"인증 오류: {message}")
    send_admin_alert(f"API 키 확인 필요: {message}")
    return {"error_type": "auth", "message": "시스템 오류입니다", "retry": False}

def handle_payment_error(message):
    logger.warning(f"결제 오류: {message}")
    return {"error_type": "payment", "message": message, "retry": False}

def handle_order_error(message):
    logger.info(f"주문 오류: {message}")
    return {"error_type": "order", "message": message, "retry": False}

def handle_amount_error(message):
    logger.info(f"금액 오류: {message}")
    return {"error_type": "amount", "message": message, "retry": False}

def handle_unknown_error(error_code, message):
    logger.error(f"알 수 없는 에러 [{error_code}]: {message}")
    return {"error_type": "unknown", "message": "알 수 없는 오류입니다", "retry": True}

    구조화된 로깅

    import logging
import json
from datetime import datetime

class TossPaymentsErrorLogger:
    def __init__(self):
        self.logger = logging.getLogger("toss_payments")

    def log_error(self, error, context=None):
        """에러 로깅"""
        error_data = {
            "timestamp": datetime.now().isoformat(),
            "error_type": type(error).__name__,
            "error_message": str(error),
            "context": context or {}
        }

        if isinstance(error, APIError):
            error_data.update({
                "error_code": error.error_code,
                "status_code": error.status_code,
                "response_data": error.response_data
            })

        self.logger.error(json.dumps(error_data, ensure_ascii=False))

    def log_retry(self, attempt, max_retries, error):
        """재시도 로깅"""
        retry_data = {
            "timestamp": datetime.now().isoformat(),
            "event": "retry_attempt",
            "attempt": attempt,
            "max_retries": max_retries,
            "error": str(error)
        }

        self.logger.warning(json.dumps(retry_data, ensure_ascii=False))

# 전역 에러 로거
error_logger = TossPaymentsErrorLogger()

def process_payment_with_logging(payment_key, order_id, amount):

        "payment_key": payment_key,
        "order_id": order_id,
        "amount": amount
    }

        payment = client.payments.confirm(

        )

    except TossPaymentsError as e:
        error_logger.log_error(e, context)
        return {"success": False, "error": str(e)}

    에러 메트릭 수집

    from collections import defaultdict, Counter
from datetime import datetime, timedelta

class ErrorMetrics:
    def __init__(self):
        self.error_counts = defaultdict(int)
        self.error_times = defaultdict(list)

    def record_error(self, error_type, error_code=None):
        """에러 발생 기록"""
        key = f"{error_type}:{error_code}" if error_code else error_type
        self.error_counts[key] += 1
        self.error_times[key].append(datetime.now())

    def get_error_rate(self, error_type, time_window_minutes=60):
        """시간대별 에러율 계산"""
        cutoff_time = datetime.now() - timedelta(minutes=time_window_minutes)
        recent_errors = [
            t for t in self.error_times[error_type] 
            if t > cutoff_time
        ]
        return len(recent_errors)

    def get_top_errors(self, limit=10):
        """가장 빈번한 에러 목록"""
        return Counter(self.error_counts).most_common(limit)

# 전역 메트릭 수집기
metrics = ErrorMetrics()

def monitored_api_call(func, *args, **kwargs):
    """API 호출 모니터링 데코레이터"""
    try:
        return func(*args, **kwargs)

    except APIError as e:
        metrics.record_error("api_error", e.error_code)

        # 에러율이 높으면 알림
        if metrics.get_error_rate("api_error") > 10:  # 1시간에 10회 이상
            send_alert("API 에러율 급증")

        raise
    except Exception as e:
        metrics.record_error(type(e).__name__)
        raise
