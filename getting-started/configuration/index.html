    설정

    TossPayments SDK 사용을 위한 설정 방법을 안내합니다.

    API 키 발급

    1. 토스페이먼츠 개발자센터 가입

    토스페이먼츠 개발자센터에서 회원가입 후 상점을 등록하세요.

    2. API 키 확인

    개발자센터 > 연동 관리 > API 키에서 확인할 수 있습니다:

    • 테스트 시크릿 키: test_sk_...로 시작
    • 라이브 시크릿 키: live_sk_...로 시작

    보안 주의사항

    • 시크릿 키는 절대 클라이언트 사이드에 노출하지 마세요
    • GitHub 등 공개 저장소에 업로드하지 마세요
    • 환경변수 또는 보안 저장소에서 관리하세요

    기본 클라이언트 설정

    환경변수 사용 (권장)

    # .env 파일 또는 시스템 환경변수
export TOSS_SECRET_KEY="test_sk_your_secret_key_here"

    import os
from tosspayments_server_sdk import Client

# 환경변수에서 키 가져오기
secret_key = os.getenv("TOSS_SECRET_KEY")
client = Client(secret_key=secret_key)

    직접 설정

    from tosspayments_server_sdk import Client

# 테스트 환경
client = Client(secret_key="test_sk_your_test_key")

# 운영 환경  
client = Client(secret_key="live_sk_your_live_key")

    고급 설정 옵션

    타임아웃 설정

    client = Client(
    secret_key="test_sk_your_key",
    timeout=60  # 60초 (기본값: 30초)
)

    재시도 설정

    client = Client(
    secret_key="test_sk_your_key",
    max_retries=5  # 최대 5회 재시도 (기본값: 3회)
)

    API 버전 지정

    client = Client(
    secret_key="test_sk_your_key",
    api_version="v1"  # API 버전 (기본값: "v1")
)

    전체 설정 예시

    client = Client(
    secret_key=os.getenv("TOSS_SECRET_KEY"),
    api_version="v1",
    timeout=45,
    max_retries=3
)

    환경별 설정 관리

    개발/테스트/운영 환경 분리

    import os

class Config:
    def __init__(self):
        self.environment = os.getenv("ENVIRONMENT", "development")

        if self.environment == "production":
            self.toss_secret_key = os.getenv("TOSS_LIVE_SECRET_KEY")
        else:
            self.toss_secret_key = os.getenv("TOSS_TEST_SECRET_KEY")

    @property
    def is_production(self):
        return self.environment == "production"

# 사용법
config = Config()
client = Client(secret_key=config.toss_secret_key)

if config.is_production:
    print("🔴 운영 환경에서 실행 중")
else:
    print("🟡 테스트 환경에서 실행 중")

    Django 설정 예시

    # settings.py
import os

TOSS_SECRET_KEY = os.getenv("TOSS_SECRET_KEY")
TOSS_TIMEOUT = int(os.getenv("TOSS_TIMEOUT", "30"))

# views.py
from django.conf import settings
from tosspayments_server_sdk import Client

client = Client(
    secret_key=settings.TOSS_SECRET_KEY,
    timeout=settings.TOSS_TIMEOUT
)

    Flask 설정 예시

    # config.py
import os

class Config:
    TOSS_SECRET_KEY = os.getenv("TOSS_SECRET_KEY")
    TOSS_TIMEOUT = int(os.getenv("TOSS_TIMEOUT", "30"))

# app.py
from flask import Flask
from tosspayments_server_sdk import Client

app = Flask(__name__)
app.config.from_object("config.Config")

client = Client(
    secret_key=app.config["TOSS_SECRET_KEY"],
    timeout=app.config["TOSS_TIMEOUT"]
)

    모드 확인

    클라이언트가 어떤 모드로 실행 중인지 확인:

    client = Client(secret_key="test_sk_your_key")

# 모드 확인
if client.is_test_mode:
    print("테스트 모드")
elif client.is_live_mode:
    print("운영 모드")

# 안전한 결제 처리
def process_payment(payment_key, order_id, amount):
    if client.is_test_mode:
        print("⚠️ 테스트 결제 진행")

    return client.payments.confirm(
        payment_key=payment_key,
        order_id=order_id,
        amount=amount
    )

    설정 검증

    클라이언트 설정이 올바른지 확인:

    from tosspayments_server_sdk.exceptions import ValidationError

try:
    client = Client(secret_key="invalid_key")
except ValidationError as e:
    print(f"설정 오류: {e}")

    보안 모범 사례

    1. 환경변수 사용

    # .env 파일 (절대 Git에 커밋하지 마세요)
TOSS_SECRET_KEY=test_sk_your_secret_key_here

    # python-dotenv 사용
from dotenv import load_dotenv
load_dotenv()

client = Client(secret_key=os.getenv("TOSS_SECRET_KEY"))

    2. AWS Systems Manager Parameter Store

    import boto3

def get_secret_key():
    ssm = boto3.client('ssm')
    response = ssm.get_parameter(
        Name='/myapp/toss/secret_key',
        WithDecryption=True
    )
    return response['Parameter']['Value']

client = Client(secret_key=get_secret_key())

    3. Docker Secrets

    # docker-compose.yml
services:
  app:
    image: myapp
    secrets:
      - toss_secret_key

secrets:
  toss_secret_key:
    external: true

    문제 해결

    일반적인 설정 오류

    1. 잘못된 API 키 형식 

      # ❌ 잘못된 형식
client = Client(secret_key="wrong_format_key")

# ✅ 올바른 형식
client = Client(secret_key="test_sk_or_live_sk_format")

    2. 환경변수 미설정 

      secret_key = os.getenv("TOSS_SECRET_KEY")
if not secret_key:
    raise ValueError("TOSS_SECRET_KEY 환경변수가 설정되지 않음")

client = Client(secret_key=secret_key)

    3. 네트워크 타임아웃 

      # 타임아웃 늘리기
client = Client(
    secret_key="your_key",
    timeout=60  # 기본 30초에서 60초로 증가
)
